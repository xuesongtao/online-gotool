feature introduction

  1. Table creation sql to json supporting multiple databases, including (pg, oracle, mysql) and other databases;table-building statements,table-building statements,table-building statements(important statements are said 3 times)

CREATE TABLE if not exists user (
    id int NOT NULL AUTO_INCREMENT,
    name varchar(10) NOT NULL COMMENT 'name ',
    gender tinyint NOT NULL DEFAULT 0 COMMENT 'Gender 0 -unknown 1 -male 2 -female ',
    age int NOT NULL COMMENT 'age ',
    created_date datetime DEFAULT CURRENT_TIMESTAMP,
    updated_date datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (id)
);
